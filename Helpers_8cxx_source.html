32vector<string> split_string(const string& argString, char delimiter, bool onlyFirst)
33{
34 vector<string> arguments;
35 if (onlyFirst) {
36 auto delimiterPos = argString.find(delimiter);
37 arguments.push_back(argString.substr(0, delimiterPos));
38 if (delimiterPos != string::npos) {
39 arguments.push_back(argString.substr(delimiterPos + 1));
40 }
41 } else {
42 string curArg;
43 std::istringstream argStream(argString);
44 while (std::getline(argStream, curArg, delimiter)) {
45 arguments.push_back(curArg);
46 }
47 }
48
49 return arguments;
50}
