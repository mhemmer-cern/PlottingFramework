30inline bool str_contains(const std::string& str, const std::string& substr, bool reverseSearch = false)
31{
32 if (reverseSearch) {
33 return (str.rfind(substr) != string::npos);
34 } else {
35 return (str.find(substr) != string::npos);
36 }
37}

61template <typename... Ts>
62string tuple_to_string(const std::tuple<Ts...>& items)
63{
64 string itemString;
65 std::apply([&](auto&&... item) { ((itemString += std::to_string(item) + ","), ...); }, items);
66 itemString.pop_back();
67 return itemString;
68}

70template <typename T>
71string vector_to_string(vector<T> items)
72{
73 string itemString;
74 for (auto& item : items) {
75 if constexpr (is_tuple<T>::value) {
76 itemString += tuple_to_string(item);
77 if (&item != &items.back()) itemString += ";";
78 } else {
79 itemString += std::to_string(item);
80 if (&item != &items.back()) itemString += ",";
81 }
82 }
83 return itemString;
84}

86template <typename T>
87T string_to_type(const string& str)
88{
89 if constexpr (std::is_same_v<double_t, T>) {
90 return std::stod(str);
91 } else if constexpr (std::is_same_v<float_t, T>) {
92 return std::stof(str);
93 } else {
94 return std::stoi(str);
95 }
96}

98template <typename... Ts>
99std::tuple<Ts...> string_to_tuple(string itemString)
100{
101 // split string
102 string curItemStr;
103 std::istringstream stream(itemString);
104 string numbers[sizeof...(Ts)];
105 uint8_t i = 0;
106 while (std::getline(stream, curItemStr, ',')) {
107 numbers[i] = curItemStr;
108 ++i;
109 }
110
111 // ugly hack...
112 if constexpr (sizeof...(Ts) == 3) {
113 return {string_to_type<uint8_t>(numbers[0]), string_to_type<double_t>(numbers[1]), string_to_type<double_t>(numbers[2])};
114 } else {
115 return {string_to_type<float_t>(numbers[0]), string_to_type<float_t>(numbers[1]), string_to_type<float_t>(numbers[2]), string_to_type<float_t>(numbers[3])};
116 }
117}

119template <typename T>
120std::vector<T> string_to_vector(string itemString)
121{
122 // safety in case user put some blank spaces between numbers
123 std::remove_if(itemString.begin(), itemString.end(), ::isspace);
124 vector<T> items;
125
126 string curItemStr;
127 std::istringstream stream(itemString);
128 if constexpr (is_tuple<T>::value) {
129 while (std::getline(stream, curItemStr, ';')) {
130 // ugly hack...
131 if constexpr (std::is_same_v<T, tuple<uint8_t, double_t, double_t>>) {
132 items.push_back(string_to_tuple<uint8_t, double_t, double_t>(curItemStr));
133 } else {
134 items.push_back(string_to_tuple<float_t, float_t, float_t, float_t>(curItemStr));
135 }
136 }
137 } else {
138 while (std::getline(stream, curItemStr, ',')) {
139 items.push_back(string_to_type<T>(curItemStr));
140 }
141 }
142 return items;
143}

145template <typename T>
146void put_in_tree(ptree& tree, const optional<T>& var, const string& label)
147{
148 if constexpr (is_vector<T>{}) // vectors are stored as comma separated strings
149 {
150 if (var) tree.put(label, vector_to_string(*var));
151 } else if constexpr (std::is_enum<T>::value) // in case using enum types of the framework
152 {
153 if (var) tree.put(label, static_cast<typename std::underlying_type<T>::type>(*var));
154 } else {
155 if (var) tree.put(label, *var);
156 }
157}

159template <typename T>
160void read_from_tree(const ptree& tree, optional<T>& var, const string& label)
161{
162 if constexpr (is_vector<T>{}) // vectors are stored as comma separated strings
163 {
164 if (auto tmp = tree.get_optional<string>(label))
165 var = string_to_vector<typename T::value_type>(*tmp);
166 } else if constexpr (std::is_enum<T>::value) // in case using enum types of the framework
167 {
168 if (auto tmp = tree.get_optional<typename std::underlying_type<T>::type>(label))
169 var = static_cast<T>(*tmp);
170 } else {
171 if (auto tmp = tree.get_optional<T>(label)) var = *tmp;
172 }
173}

175template <typename T>
176optional<T> pick(int32_t i, const optional<vector<T>>& vec)
177{
178 if (!vec || vec->empty()) return std::nullopt;
179 return optional((*vec)[(i - 1) % vec->size()]);

182template <typename T, typename... Ts>
183const optional<T>& get_first(const optional<T>& property, const Ts&... properties)
184{
185 for (const auto ptr : {&property, &properties...}) {
186 if (*ptr) return *ptr;
187 }
188 // if all properties are std::nullopt, simply return reference to the first
189 return property;
190}
191
192template <typename T, typename... Ts>
193T get_first_or(const T& fallback, const optional<T>& property, const Ts&... properties)
194{
195 if (const auto& match = get_first(property, properties...)) {
196 return *match;
197 } else {
198 return fallback;
199 }
200}
