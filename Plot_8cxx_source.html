36Plot::Plot(const string& name, const string& figureGroupAndCategory, const optional<string>& plotTemplateName) : Plot()
37{
38 if (str_contains(figureGroupAndCategory, ".")) {
39 ERROR("Figure group must not contain '.'!");
40 std::exit(EXIT_FAILURE);
41 }
42 mName = name;
43 mFigureGroup = figureGroupAndCategory;
44 mPlotTemplateName = plotTemplateName;
45
46 // in case category was specified via figureGroup/my/category/tree
47 if (auto subPathPos = figureGroupAndCategory.find("/"); subPathPos != string::npos) {
48 mFigureGroup = figureGroupAndCategory.substr(0, subPathPos);
49 mFigureCategory = figureGroupAndCategory.substr(subPathPos + 1);
50 }
51 UpdateUniqueName();
52}

69/**
70 * Constructor from property tree.
71 */
72//**************************************************************************************************
73Plot::Plot(const ptree& plotTree)
74{
75 try {
76 mName = plotTree.get<string>("name");
77 mFigureGroup = plotTree.get<string>("figure_group");
78 } catch (...) {
79 ERROR("Could not construct data from ptree.");
80 std::exit(EXIT_FAILURE);
81 }
82 read_from_tree(plotTree, mFigureCategory, "figure_category");
83 read_from_tree(plotTree, mPlotTemplateName, "plot_template_name");
84 read_from_tree(plotTree, mPlotDimensions.width, "width");
85 read_from_tree(plotTree, mPlotDimensions.height, "height");
86 read_from_tree(plotTree, mPlotDimensions.fixAspectRatio, "fix_aspect_ratio");
87 read_from_tree(plotTree, mFill.color, "fill_color");
88 read_from_tree(plotTree, mFill.style, "fill_style");
89 read_from_tree(plotTree, mFill.scale, "fill_opacity");
90
91 // loop over all pads defined in property tree
92 for (auto& pad : plotTree) {
93 if (str_contains(pad.first, "PAD")) {
94 uint8_t padID = std::stoi(pad.first.substr(pad.first.find("_") + 1));
95 mPads[padID] = Pad(pad.second);
96 }
97 }
98 UpdateUniqueName();
99}

112/**
113 * Get representation of plot as property tree.
114 */
115//**************************************************************************************************
116ptree Plot::GetPropertyTree() const
117{
118 ptree plotTree;
119 plotTree.put("name", mName);
120 plotTree.put("figure_group", mFigureGroup);
121 put_in_tree(plotTree, mFigureCategory, "figure_category");
122 put_in_tree(plotTree, mPlotTemplateName, "plot_template_name");
123 put_in_tree(plotTree, mPlotDimensions.width, "width");
124 put_in_tree(plotTree, mPlotDimensions.height, "height");
125 put_in_tree(plotTree, mPlotDimensions.fixAspectRatio, "fix_aspect_ratio");
126 put_in_tree(plotTree, mFill.color, "fill_color");
127 put_in_tree(plotTree, mFill.style, "fill_style");
128 put_in_tree(plotTree, mFill.scale, "fill_opacity");
129
130 for (auto& [padID, pad] : mPads) {
131 plotTree.put_child("PAD_" + std::to_string(padID), pad.GetPropertyTree());
132 }
133 return plotTree;
134}

136//**************************************************************************************************
137/**
138 * Make a deep copy of the Plot where also data and boxes are copied.
139 */
140//**************************************************************************************************
141Plot Plot::Clone() const
142{
143 Plot newPlot(*this);
144 for (auto& [padID, pad] : newPlot.GetPads()) {
145 std::transform(newPlot[padID].GetData().begin(), newPlot[padID].GetData().end(),
146 newPlot[padID].GetData().begin(),
147 [](auto& data_ptr) { return data_ptr->Clone(); });
148
149 std::transform(newPlot[padID].GetLegendBoxes().begin(), newPlot[padID].GetLegendBoxes().end(),
150 newPlot[padID].GetLegendBoxes().begin(), [](auto& legend_ptr) {
151 return std::make_shared<Plot::Pad::LegendBox>(*legend_ptr);
152 });
153
154 std::transform(newPlot[padID].GetTextBoxes().begin(), newPlot[padID].GetTextBoxes().end(),
155 newPlot[padID].GetTextBoxes().begin(),
156 [](auto& text_ptr) { return std::make_shared<Plot::Pad::TextBox>(*text_ptr); });
157 }
158 return newPlot;
159}

257//**************************************************************************************************
258/**
259 * Apply all settings from plot on top of this plot.
260 */
261//**************************************************************************************************
262void Plot::operator+=(const Plot& plot)
263{
264 mName = plot.mName;
265 mFigureGroup = plot.mFigureGroup;
266 mFigureCategory = plot.mFigureCategory;
267 if (plot.mFigureCategory) mFigureCategory = plot.mFigureCategory;
268 mPlotTemplateName = plot.mPlotTemplateName;
269
270 if (plot.mPlotDimensions.width) mPlotDimensions.width = plot.mPlotDimensions.width;
271 if (plot.mPlotDimensions.height) mPlotDimensions.height = plot.mPlotDimensions.height;
272 if (plot.mPlotDimensions.fixAspectRatio) mPlotDimensions.fixAspectRatio = plot.mPlotDimensions.fixAspectRatio;
273
274 if (plot.mFill.color) mFill.color = plot.mFill.color;
275 if (plot.mFill.style) mFill.style = plot.mFill.style;
276 if (plot.mFill.scale) mFill.scale = plot.mFill.scale;
277
278 for (auto& [padID, pad] : plot.mPads) {
279 mPads[padID]; // initializes the pad in case it was not yet defined in this plot
280 mPads[padID] += pad;
281 }
282 UpdateUniqueName();
283}

759

1514//**************************************************************************************************
1515/**
1516 * Name suffix encoding the details of projections.
1517 */
1518//**************************************************************************************************
1519std::string Plot::Pad::Data::proj_info_t::GetNameSuffix() const
1520{
1521 std::string nameSuffix = (isProfile && *isProfile) ? "_Prof{" : "_Proj{";
1522 for (auto dim : dims) {
1523 nameSuffix += std::to_string(dim);
1524 }
1525 nameSuffix += "}";
1526 if (!ranges.empty()) {
1527 nameSuffix += "_Range";
1528 for (auto& range : ranges) {
1529 nameSuffix += "{";
1530 nameSuffix += std::to_string(std::get<0>(range)) + ":";
1531 if (isUserCoord && *isUserCoord) {
1532 nameSuffix += std::to_string(std::get<1>(range)) + "," + std::to_string(std::get<2>(range));
1533 } else {
1534 nameSuffix += std::to_string(static_cast<int>(std::get<1>(range))) + "," + std::to_string(static_cast<int>(std::get<2>(range)));
1535 }
1536 nameSuffix += "}";
1537 }
1538 }
1539 if (isUserCoord && *isUserCoord) {
1540 nameSuffix += "_User";
1541 }
1542 return nameSuffix;
1543}
